  **Frontend-Backend Integration Wiring:**
  ```typescript
  // In LoginPage.tsx
  import { authApi } from '@/api/authApi';
  import { useAuthStore } from '@/stores/authStore';
  import { useNavigate } from 'react-router-dom';

  // Form submission handler
  const onSubmit = async (data: LoginFormData) => {
    try {
      // Step 1: Call authApi.login which uses apiClient
      const response = await authApi.login(data.email, data.password);

      // Step 2: Store tokens in Zustand store
      useAuthStore.getState().setTokens(response.accessToken, response.refreshToken);
      useAuthStore.getState().setUser({
        id: response.userId,
        email: data.email,
        displayName: null,
        avatarUrl: null,
        createdAt: new Date().toISOString(),
        listingCount: 0
      });

      // Step 3: Navigate to home
      navigate('/');
    } catch (error) {
      // Error toast shown via catch block
      toast.error("Invalid email or password. Please try again.");
    }
  };
  ```
